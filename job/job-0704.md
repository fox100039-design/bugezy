```ts
import { t, getUILang, UILang } from './i18n';

let currentUILang: UILang = 'zh';

// PM-137 的語言選擇改為同時控制 UI
langSelect.addEventListener('change', async () => {

function applyTranslations() {

}
```

```ts
});
```

```ts
  const { data: existing } = await supa(env)
    .from('payments')
    .select('status')
    .eq('merchant_trade_no', tradeNo)
    .maybeSingle();
  if (existing?.status === 'paid') {
    return new Response('1|OK'); // 冪等：不重複處理
  }

  // 3. 金額比對（月費應為 80）
  const amount = parseInt(params.TradeAmt as string, 10);
  if (amount !== 80) {
    console.error(`ECPay callback amount mismatch: expected 80, got ${amount}`);
    return new Response('1|OK'); // 金額不對不授權，但回 1|OK 讓綠界停止重送
  }

  // 4. 處理付款（現有邏輯）
  if (params.RtnCode === '1') {
    // ... 更新 user plan ...
    
    // 5. 記錄到 payments 表
    await supa(env).from('payments').upsert({
      merchant_trade_no: tradeNo,
      user_id: params.CustomField1,
      payment_type: 'monthly',
      amount: amount,
      rtn_code: params.RtnCode,
      status: 'paid',
      raw_callback: params,
      paid_at: new Date().toISOString(),
    });
  } else {
    // 付款失敗也記錄
    await supa(env).from('payments').upsert({
      merchant_trade_no: tradeNo,
      user_id: params.CustomField1 || '',
      payment_type: 'monthly',
      amount: amount,
      rtn_code: params.RtnCode,
      status: 'failed',
      raw_callback: params,
    });
  }

  return new Response('1|OK');
}
```

```ts
async function handleLogout(request: Request, env: Env): Promise<Response> {
  const auth = request.headers.get('Authorization');
  if (!auth) return json({ ok: true });
  const token = auth.replace('Bearer ', '').trim();
  if (!token) return json({ ok: true });

  // 從 sessions 表刪除
  await supa(env).from('sessions').delete().eq('session_token', token);
  return json({ ok: true });
}
```
