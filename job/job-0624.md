#### §1 差異：校正 vs 精簡

| | 🔧 AI 校正（新） | 🤖 AI 精簡（現有） |
|---|---|---|
| 目的 | 修錯字，保留原意 | 移除廢話，濃縮重點 |
| 輸出長度 | 跟原文差不多 | 比原文短很多 |
| 使用時機 | 語音有錯字時 | 語音太囉嗦時 |
| 可疊加 | 先校正 → 再精簡 | ✅ |

#### §6 popup.ts 登入邏輯

```ts
const SESSION_KEY = 'bugezy:session';

interface Session {
  user_id: string;
  email: string;
  name: string;
  avatar_url: string;
  session_token: string;
}

// 檢查是否已登入
async function checkAuth(): Promise<Session | null> {
  const r = await chrome.storage.local.get(SESSION_KEY);
  return r[SESSION_KEY] || null;
}

// Google 登入
async function googleLogin() {
  return new Promise<string>((resolve, reject) => {
    chrome.identity.getAuthToken({ interactive: true }, (token) => {
      if (chrome.runtime.lastError || !token) {
        reject(chrome.runtime.lastError?.message || 'login failed');
      } else {
        resolve(token);
      }
    });
  });
}

// 送 token 給 server 驗證
async function authenticate(googleToken: string): Promise<Session> {
  const res = await fetch(`${API_BASE}/api/auth/google`, {
    method: 'POST',
    headers: { 'Content-Type': 'application/json' },
    body: JSON.stringify({ token: googleToken }),
  });
  if (!res.ok) throw new Error('auth failed');
  return res.json();
}

// 登入按鈕
googleLoginBtn.addEventListener('click', async () => {
  try {
    googleLoginBtn.disabled = true;
    googleLoginBtn.textContent = '登入中...';
    const googleToken = await googleLogin();
    const session = await authenticate(googleToken);
    await chrome.storage.local.set({ [SESSION_KEY]: session });
    showMainView(session);
  } catch (err) {
    googleLoginBtn.textContent = '登入失敗，重試';
    googleLoginBtn.disabled = false;
  }
});

// 登出
logoutBtn.addEventListener('click', async () => {
  await chrome.storage.local.remove(SESSION_KEY);
  chrome.identity.clearAllCachedAuthTokens(() => {});
  showLoginView();
});

// 頁面載入
const session = await checkAuth();
if (session) {
  showMainView(session);
} else {
  showLoginView();
}
```

```ts
  if (!googleRes.ok) return json({ error: 'invalid google token' }, 401);
  
  const googleUser = await googleRes.json() as {
    id: string; email: string; name: string; picture: string;
  };
  
  // 查/建 user
  const supabase = supa(env);
  let { data: user } = await supabase
    .from('users')
    .select('*')
    .eq('email', googleUser.email)
    .single();
  
  if (!user) {
    // 新用戶
    const { data: newUser } = await supabase
      .from('users')
      .insert({
        email: googleUser.email,
        name: googleUser.name,
        avatar_url: googleUser.picture,
      })
      .select()
      .single();
    user = newUser;
  } else {
    // 更新 last_login
    await supabase
      .from('users')
      .update({ last_login_at: new Date().toISOString(), name: googleUser.name, avatar_url: googleUser.picture })
      .eq('user_id', user.user_id);
  }
  
  // 簡單的 session token（用 user_id + timestamp 的 hash）
  // 生產環境應改用 JWT，但 MVP 先用簡單方式
  const session_token = btoa(`${user.user_id}:${Date.now()}`);
  
  return json({
    user_id: user.user_id,
    email: user.email,
    name: user.name,
    avatar_url: user.avatar_url,
    session_token,
  });
}
```
