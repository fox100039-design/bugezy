#### §3 Server — GET /api/user/plan

查詢用戶方案 + 剩餘用量：

```ts
if (url.pathname === '/api/user/plan' && request.method === 'GET') {
  const userId = getUserIdFromHeader(request); // 從 Authorization header 讀
  if (!userId) return json({ error: 'unauthorized' }, 401);
  
  const { data: user } = await supa(env)
    .from('users')
    .select('plan, recording_count, rewind_count, mcp_count, usage_reset_at')
    .eq('user_id', userId)
    .maybeSingle();
  
  if (!user) return json({ error: 'user not found' }, 404);
  
  // 每月重置檢查
  const resetAt = new Date(user.usage_reset_at);
  const now = new Date();
  if (now.getMonth() !== resetAt.getMonth() || now.getFullYear() !== resetAt.getFullYear()) {
    // 新的一個月，重置計數
    await supa(env)
      .from('users')
      .update({ recording_count: 0, rewind_count: 0, mcp_count: 0, usage_reset_at: now.toISOString() })
      .eq('user_id', userId);
    user.recording_count = 0;
    user.rewind_count = 0;
    user.mcp_count = 0;
  }
  
  const isPaid = user.plan === 'paid';
  
  return json({
    plan: user.plan,
    limits: isPaid ? null : {
      recording: { used: user.recording_count, max: FREE_LIMITS.recording },
      rewind: { used: user.rewind_count, max: FREE_LIMITS.rewind },
      mcp: { used: user.mcp_count, max: FREE_LIMITS.mcp },
    },
  });
}
```

#### §4 Server — POST /api/user/usage

錄製/回溯/MCP 使用時遞增計數，免費版超過限制回 403：

```ts
if (url.pathname === '/api/user/usage' && request.method === 'POST') {
  const userId = getUserIdFromHeader(request);
  if (!userId) return json({ error: 'unauthorized' }, 401);
  
  const { type } = await request.json() as { type: 'recording' | 'rewind' | 'mcp' };
  
  const { data: user } = await supa(env)
    .from('users')
    .select('plan, recording_count, rewind_count, mcp_count')
    .eq('user_id', userId)
    .maybeSingle();
  
  if (!user) return json({ error: 'user not found' }, 404);
  
  // 付費版不限制
  if (user.plan === 'paid') {
    return json({ ok: true, unlimited: true });
  }
  
  // 免費版檢查限制
  const countField = `${type}_count` as keyof typeof user;
  const currentCount = (user[countField] as number) || 0;
  const limit = FREE_LIMITS[type];
  
  if (currentCount >= limit) {
    return json({
      error: 'limit_reached',
      message: `免費版每月限 ${limit} 次${type === 'recording' ? '錄製' : type === 'rewind' ? '回溯' : 'MCP 查詢'}，升級付費版解鎖無限次`,
      used: currentCount,
      max: limit,
    }, 403);
  }
  
  // 遞增計數
  await supa(env)
    .from('users')
    .update({ [countField]: currentCount + 1 })
    .eq('user_id', userId);
  
  return json({ ok: true, used: currentCount + 1, max: limit });
}
```

---

#### §5 getUserIdFromHeader 函式

```ts
function getUserIdFromHeader(request: Request): string | null {
  const auth = request.headers.get('Authorization');
  if (!auth) return null;
  try {
    // session_token 格式：base64(user_id:timestamp)
    const decoded = atob(auth.replace('Bearer ', ''));
    return decoded.split(':')[0] || null;
  } catch {
    return null;
  }
}
```
