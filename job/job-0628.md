#### §2 CheckMacValue 計算函式

```ts
function generateCheckMacValue(params: Record<string, string>, hashKey: string, hashIV: string): string {
  // 1. 按 key 字母排序
  const sorted = Object.keys(params).sort((a, b) => a.toLowerCase().localeCompare(b.toLowerCase()));

  // 2. 組成 query string
  let raw = `HashKey=${hashKey}`;
  for (const key of sorted) {
    raw += `&${key}=${params[key]}`;
  }
  raw += `&HashIV=${hashIV}`;

  // 3. URL encode（綠界規定用 .NET 的 URLEncode 規則）
  let encoded = encodeURIComponent(raw).toLowerCase();
  // 綠界特殊字元還原
  encoded = encoded.replace(/%2d/g, '-').replace(/%5f/g, '_').replace(/%2e/g, '.').replace(/%21/g, '!').replace(/%2a/g, '*').replace(/%28/g, '(').replace(/%29/g, ')').replace(/%20/g, '+');

  // 4. SHA256 → 大寫
  const encoder = new TextEncoder();
  const data = encoder.encode(encoded);
  // Cloudflare Workers 用 crypto.subtle
  // 注意：這需要改成 async
  // 用同步的方式需要引入 crypto module
}
```

⚠ Cloudflare Workers 的 SHA256 是 async（`crypto.subtle.digest`）。generateCheckMacValue 要改成 async。

```ts
async function generateCheckMacValue(params: Record<string, string>, hashKey: string, hashIV: string): Promise<string> {
  // ...步驟 1-3 同上...
  const data = new TextEncoder().encode(encoded);
  const hashBuffer = await crypto.subtle.digest('SHA-256', data);
  const hashArray = Array.from(new Uint8Array(hashBuffer));
  return hashArray.map(b => b.toString(16).padStart(2, '0')).join('').toUpperCase();
}
```

#### 新增 POST /api/ecpay/period-callback

每月自動扣款後，綠界會通知這個 URL。需要：
1. 驗證 CheckMacValue
2. 如果扣款成功（RtnCode=1）→ 續期（plan 維持 'paid'）
3. 如果扣款失敗 → 降級（plan 改回 'free'）

```ts
if (url.pathname === '/api/ecpay/period-callback' && request.method === 'POST') {
  const formData = await request.formData();
  const params: Record<string, string> = {};
  formData.forEach((val, key) => { params[key] = String(val); });

  // 驗證 CheckMacValue
  const receivedMac = params.CheckMacValue;
  const paramsForCheck = { ...params };
  delete paramsForCheck.CheckMacValue;
  const expectedMac = await generateCheckMacValue(paramsForCheck, env.ECPAY_HASH_KEY, env.ECPAY_HASH_IV);

  if (receivedMac !== expectedMac) {
    return new Response('0|ErrorMessage=CheckMacValue Error');
  }

  const userId = params.CustomField1;
  if (!userId) {
    return new Response('1|OK');
  }

  if (params.RtnCode === '1') {
    // 扣款成功 → 維持 paid
    await supa(env)
      .from('users')
      .update({ plan: 'paid', last_login_at: new Date().toISOString() })
      .eq('user_id', userId);
  } else {
    // 扣款失敗 → 降級為 free
    await supa(env)
      .from('users')
      .update({ plan: 'free' })
      .eq('user_id', userId);
  }

  return new Response('1|OK');
}
```
