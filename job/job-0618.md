#### §2 edit-report.ts 時間軸標記邏輯

```ts
import { Replayer } from '@rrweb/replay';
import '@rrweb/replay/dist/style.css';

interface Marker {
  time_sec: number;
  note: string;
}

const markers: Marker[] = [];
let replayer: Replayer | null = null;
let duration = 0;
let playing = false;

// 初始化 mini player（如果有 rrweb 資料）
function initMiniPlayer(events: unknown[]) {
  const container = document.getElementById('miniPlayer');
  if (!container || events.length < 2) {
    document.getElementById('markerSection')!.style.display = 'none';
    return;
  }

  replayer = new Replayer(events as any, {
    root: container,
    skipInactive: true,
    showWarning: false,
  });

  const meta = replayer.getMetaData();
  duration = meta.totalTime || 0;
  const seekBar = document.getElementById('markerSeek') as HTMLInputElement;
  seekBar.max = String(duration);

  // 播放/暫停
  document.getElementById('markerPlayBtn')!.addEventListener('click', () => {
    if (!replayer) return;
    if (playing) {
      replayer.pause();
      playing = false;
      document.getElementById('markerPlayBtn')!.textContent = '▶';
    } else {
      const current = replayer.getCurrentTime();
      if (current >= duration - 100) {
        replayer.play(0);
      } else {
        replayer.resume(current);
      }
      playing = true;
      document.getElementById('markerPlayBtn')!.textContent = '⏸';
      trackProgress();
    }
  });

  // seek
  seekBar.addEventListener('input', (e) => {
    const time = Number((e.target as HTMLInputElement).value);
    replayer!.play(time);
    if (!playing) setTimeout(() => replayer!.pause(time), 50);
    updateTimeDisplay(time);
  });

  // 📌 標記按鈕
  document.getElementById('addMarkerBtn')!.addEventListener('mousedown', (e) => {
    e.preventDefault(); // 不搶焦點
  });
  document.getElementById('addMarkerBtn')!.addEventListener('click', () => {
    if (!replayer) return;
    const currentTime = replayer.getCurrentTime();
    const sec = Math.floor(currentTime / 1000);
    // 暫停在當前位置
    if (playing) {
      replayer.pause();
      playing = false;
      document.getElementById('markerPlayBtn')!.textContent = '▶';
    }
    addMarker(sec);
  });
}

function addMarker(sec: number) {
  const marker: Marker = { time_sec: sec, note: '' };
  markers.push(marker);
  renderMarkers();
  // 自動 focus 到新標記的輸入框
  setTimeout(() => {
    const inputs = document.querySelectorAll('.marker-note');
    const last = inputs[inputs.length - 1] as HTMLInputElement;
    last?.focus();
  }, 50);
}

function renderMarkers() {
  const list = document.getElementById('markerList')!;
  list.innerHTML = '';
  // 按時間排序
  markers.sort((a, b) => a.time_sec - b.time_sec);
  markers.forEach((m, i) => {
    const item = document.createElement('div');
    item.className = 'marker-item';
    
    const timeBtn = document.createElement('span');
    timeBtn.className = 'marker-time';
    timeBtn.textContent = formatSec(m.time_sec);
    timeBtn.title = '點擊跳到此時間';
    timeBtn.addEventListener('click', () => {
      if (replayer) {
        replayer.play(m.time_sec * 1000);
        setTimeout(() => replayer!.pause(m.time_sec * 1000), 50);
        updateTimeDisplay(m.time_sec * 1000);
      }
    });
    
    const noteInput = document.createElement('input');
    noteInput.className = 'marker-note';
    noteInput.type = 'text';
    noteInput.placeholder = '描述這個時間點的問題...';
    noteInput.value = m.note;
    noteInput.addEventListener('input', () => { m.note = noteInput.value; });
    
    const delBtn = document.createElement('button');
    delBtn.className = 'marker-delete';
    delBtn.textContent = '✕';
    delBtn.addEventListener('click', () => {
      markers.splice(i, 1);
      renderMarkers();
    });
    
    item.appendChild(timeBtn);
    item.appendChild(noteInput);
    item.appendChild(delBtn);
    list.appendChild(item);
  });
}

function formatSec(sec: number): string {
  const m = Math.floor(sec / 60);
  const s = sec % 60;
  return `${m}:${String(s).padStart(2, '0')}`;
}

function updateTimeDisplay(ms: number) {
  const el = document.getElementById('markerTime');
  if (el) el.textContent = `${formatSec(Math.floor(ms / 1000))} / ${formatSec(Math.floor(duration / 1000))}`;
  const seekBar = document.getElementById('markerSeek') as HTMLInputElement;
  seekBar.value = String(ms);
}

function trackProgress() {
  if (!playing || !replayer) return;
  const current = replayer.getCurrentTime();
  updateTimeDisplay(current);
  if (current < duration) {
    requestAnimationFrame(trackProgress);
  } else {
    playing = false;
    document.getElementById('markerPlayBtn')!.textContent = '▶';
  }
}
```
