#### §2 Server：新增日票付款端點

```
POST /api/day-pass/create
```

- 需要登入（從 auth header 取 user_id）
- 建立綠界一次性付款訂單
- 訂單金額 NT$20
- TradeDesc：「BugEzy 日票（24小時）」
- 付款方式：ALL（信用卡 + ATM + 超商）

```ts
async function handleDayPassCreate(request: Request, env: Env): Promise<Response> {
  // 1. 驗證登入
  const user = await getAuthUser(request, env);
  if (!user) return json({ error: '請先登入' }, 401);

  // 2. 檢查是否已有日票或月費
  const { data } = await supabase(env)
    .from('users')
    .select('plan, day_pass_expires_at')
    .eq('user_id', user.user_id)
    .single();

  if (data?.plan === 'paid') {
    return json({ error: '您已是月費用戶，不需購買日票' }, 400);
  }
  if (data?.day_pass_expires_at && new Date(data.day_pass_expires_at) > new Date()) {
    return json({ error: '您已有有效日票，到期後才能再購買' }, 400);
  }

  // 3. 建立綠界訂單（一次性付款，非定期定額）
  const tradeNo = 'DP' + Date.now(); // 唯一訂單編號
  const params = {
    MerchantID: env.ECPAY_MERCHANT_ID,
    MerchantTradeNo: tradeNo,
    MerchantTradeDate: formatECPayDate(new Date()),
    PaymentType: 'aio',
    TotalAmount: '20',
    TradeDesc: 'BugEzy 日票（24小時無限使用）',
    ItemName: 'BugEzy 日票 NT$20',
    ReturnURL: `${getBaseUrl(request)}/api/day-pass/callback`,
    ClientBackURL: `${getBaseUrl(request)}/day-pass-success`,
    ChoosePayment: 'ALL',
    EncryptType: '1',
    CustomField1: user.user_id, // 用來在 callback 識別使用者
  };

  // 4. 產生檢查碼 + 回傳綠界付款 URL
  // （跟現有月費的 ECPay 串接邏輯一樣，複用 generateCheckMacValue）
  const checkMac = generateCheckMacValue(params, env.ECPAY_HASH_KEY, env.ECPAY_HASH_IV);
  params.CheckMacValue = checkMac;

  // 5. 回傳 form HTML（auto-submit 到綠界）
  return new Response(buildECPayFormHTML(params, env.ECPAY_PAYMENT_URL), {
    headers: { 'Content-Type': 'text/html; charset=utf-8' },
  });
}
```

#### §3 Server：日票 callback

```
POST /api/day-pass/callback
```

綠界付款成功後打回來：

```ts
async function handleDayPassCallback(request: Request, env: Env): Promise<Response> {
  const body = await request.formData();
  const params = Object.fromEntries(body.entries());

  // 1. 驗證 CheckMacValue
  if (!verifyCheckMacValue(params, env.ECPAY_HASH_KEY, env.ECPAY_HASH_IV)) {
    return new Response('CheckMacValue Error', { status: 400 });
  }

  // 2. 確認付款成功（RtnCode === '1'）
  if (params.RtnCode !== '1') {
    return new Response('1|OK'); // 綠界要求回 1|OK
  }

  // 3. 取 user_id（從 CustomField1）
  const userId = params.CustomField1 as string;

  // 4. 設定日票到期時間（24 小時後）
  const expiresAt = new Date(Date.now() + 24 * 60 * 60 * 1000).toISOString();

  await supabase(env)
    .from('users')
    .update({
      day_pass_expires_at: expiresAt,
      plan: 'day_pass', // 新增 plan 值
    })
    .eq('user_id', userId);

  return new Response('1|OK');
}
```

```ts
function isActiveUser(user: { plan: string; day_pass_expires_at?: string | null }): boolean {
```

#### §4 日票有效中的 UI

```ts
function showDayPassActive(remainMs: number) {
  // 隱藏升級按鈕
  upgradeSection.style.display = 'none';
  
  // 顯示日票狀態列
  dayPassStatus.style.display = 'flex';
  dayPassStatus.innerHTML = `
    <span class="day-pass-badge">⚡ 日票</span>
    <span id="day-pass-countdown" class="countdown"></span>
  `;

  // 倒數計時器
  updateCountdown(remainMs);
  const timer = setInterval(() => {
    remainMs -= 1000;
    if (remainMs <= 0) {
      clearInterval(timer);
      // 到期：刷新顯示
      location.reload();
    }
    updateCountdown(remainMs);
  }, 1000);
}

function updateCountdown(ms: number) {
  const h = Math.floor(ms / 3600000);
  const m = Math.floor((ms % 3600000) / 60000);
  const s = Math.floor((ms % 60000) / 1000);
  const el = document.getElementById('day-pass-countdown');
  if (el) el.textContent = `剩餘 ${h}h ${m}m ${s}s`;
}
```

```ts
const PROMPTS_KEY = 'bugezy:ai-prompts';
const DEFAULT_PROMPTS = [
  '請讀取我最新的 BugEzy 報告，幫我找出問題並修復',
  '請讀取最新 BugEzy 報告，分析：\n1. 真正的 root cause\n2. 修復方案\n3. 修改哪些檔案\n4. 產生 fix plan\n請不要猜測，如果資料不足請告知需要哪些資訊',
  '請讀取我最新的截圖報告，看畫面哪裡有問題，給我 CSS/HTML 修復建議',
  '請讀取最新 BugEzy 報告，直接給我可以貼上的修復程式碼',
];

let prompts: string[] = [];
let currentIndex = 0;

// 初始化
async function initPrompts() {
  const store = await chrome.storage.local.get(PROMPTS_KEY);
  prompts = store[PROMPTS_KEY] || [...DEFAULT_PROMPTS];
  currentIndex = 0;
  renderPrompt();
}

function renderPrompt() {
  promptPreview.textContent = prompts[currentIndex].split('\n')[0]; // 第一行預覽
  promptIndex.textContent = `${currentIndex + 1}/${prompts.length}`;
}

// ◀ ▶ 切換
promptPrev.addEventListener('click', () => {
  currentIndex = (currentIndex - 1 + prompts.length) % prompts.length;
  renderPrompt();
});
promptNext.addEventListener('click', () => {
  currentIndex = (currentIndex + 1) % prompts.length;
  renderPrompt();
});

// 📋 複製（完整全文）
promptCopy.addEventListener('click', async () => {
  await navigator.clipboard.writeText(prompts[currentIndex]);
  promptCopied.style.display = 'inline-block';
  setTimeout(() => { promptCopied.style.display = 'none'; }, 2000);
});

// ✏️ 編輯
promptEdit.addEventListener('click', () => {
  promptTextarea.value = prompts[currentIndex];
  promptEditor.style.display = 'block';
});

// 💾 儲存
promptSave.addEventListener('click', async () => {
  prompts[currentIndex] = promptTextarea.value.trim();
  await chrome.storage.local.set({ [PROMPTS_KEY]: prompts });
  promptEditor.style.display = 'none';
  renderPrompt();
});

// 取消
promptCancel.addEventListener('click', () => {
  promptEditor.style.display = 'none';
});

initPrompts();
```

#### §2 自動儲存切換前的編輯

切換時如果 textarea 有改動，自動存回當前則再切：

```ts
function switchPrompt(direction: number) {
  // 如果編輯器開著，先存當前的
  if (promptEditor.style.display !== 'none') {
    const newText = promptTextarea.value.trim();
    if (newText && newText !== prompts[currentIndex].text) {
      prompts[currentIndex] = { text: newText, color: editingColor };
      chrome.storage.local.set({ [PROMPTS_KEY]: prompts });
    }
  }

  currentIndex = (currentIndex + direction + prompts.length) % prompts.length;
  renderPrompt();

  if (promptEditor.style.display !== 'none') {
    syncEditorToCurrentPrompt();
  }
}
```

```ts
const PIN_KEY = 'bugezy:prompt-pinned';
let isPinned = false;

async function initPin() {
  const store = await chrome.storage.local.get(PIN_KEY);
  isPinned = store[PIN_KEY] === true;
  updatePinUI();
}

function updatePinUI() {
  promptBody.style.display = isPinned ? 'block' : 'none';
  promptPin.classList.toggle('pinned', isPinned);
}

// 點標題列：未釘選時切換展開/收合
promptHeader.addEventListener('click', (e) => {
  // 避免按鈕點擊觸發
  if ((e.target as HTMLElement).closest('button')) return;
  if (!isPinned) {
    promptBody.style.display = promptBody.style.display === 'none' ? 'block' : 'none';
  }
});

// 📌 釘選按鈕
promptPin.addEventListener('click', async (e) => {
  e.stopPropagation();
  isPinned = !isPinned;
  await chrome.storage.local.set({ [PIN_KEY]: isPinned });
  updatePinUI();
  if (isPinned) {
    promptBody.style.display = 'block';
  }
});
```

```ts
const SETTINGS_OPEN_KEY = 'bugezy:settings-open';
const settingsHeader = document.getElementById('settings-header')!;
const settingsBody = document.getElementById('settings-body')!;
const settingsChevron = document.getElementById('settings-chevron')!;

// 初始化：讀 storage
const settingsStore = await chrome.storage.local.get(SETTINGS_OPEN_KEY);
let settingsOpen = settingsStore[SETTINGS_OPEN_KEY] === true; // 預設收合
updateSettingsUI();

settingsHeader.addEventListener('click', async () => {
  settingsOpen = !settingsOpen;
  await chrome.storage.local.set({ [SETTINGS_OPEN_KEY]: settingsOpen });
  updateSettingsUI();
});

function updateSettingsUI() {
  settingsBody.style.display = settingsOpen ? 'block' : 'none';
  settingsChevron.classList.toggle('open', settingsOpen);
}
```
