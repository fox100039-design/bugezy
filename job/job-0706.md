#### §2 Server bumpUsage：每月自動重置

bumpUsage 開頭檢查 usage_reset_at，超過 30 天就歸零：

```ts
const resetAt = new Date(u.usage_reset_at || 0);
const now = new Date();
const daysSinceReset = (now.getTime() - resetAt.getTime()) / (1000 * 60 * 60 * 24);

if (daysSinceReset >= 30) {
  await supa(env).from('users').update({
    recording_count: 0, rewind_count: 0, mcp_count: 0,
    usage_reset_at: now.toISOString(),
  }).eq('user_id', userId);
  u.recording_count = 0;
  u.rewind_count = 0;
  u.mcp_count = 0;
}
```

bumpUsage 的 select 加 `usage_reset_at`。

#### §4 Extension background.ts：回溯加用量檢查

新增 checkRewindUsage()，仿照 checkRecordingUsage()：
```ts
async function checkRewindUsage(): Promise<string | null> {
  const headers = await getAuthHeaders();
  if (!headers.Authorization) return null;
  try {
    const res = await fetch(`${API_BASE}/api/user/usage`, {
      method: 'POST', headers,
      body: JSON.stringify({ type: 'rewind' }),
    });
    if (res.status === 403) {
      const err = await res.json();
      if (err.error === 'limit_reached') return err.message || '回溯額度已用完';
    }
    return null;
  } catch { return null; }
}
```

在 startRewind() 最前面呼叫，達上限不進入回溯。

### 🟡 PM-176：CLI stderr 智慧解析 — Python traceback 結構化

**背景**：bugezy-watch 目前把 stderr 當純文字上傳，AI 要自己解析。Python 的 traceback 格式是規律的，可以自動解析成結構化資料讓 AI 一秒定位。

**目標**：CLI 端解析 Python traceback → 結構化 JSON → 上傳到 server。

---

#### §1 cli/ 新增 parse-traceback.ts

```ts
export interface StackFrame {
  file: string;
  line: number;
  function: string;
  code: string;
}

export interface ParsedError {
  type: string;           // 'KeyError', 'TypeError', etc.
  message: string;        // "'key'"
  frames: StackFrame[];   // 堆疊（從最外層到最內層）
  raw: string;            // 原始文字（備用）
  timestamp: number;
}

/**
 * 解析 Python traceback 文字 → 結構化
 * 
 * 輸入範例：
 * Traceback (most recent call last):
 *   File "app.py", line 42, in main
 *     result = process_data(data)
 *   File "utils.py", line 15, in process_data
 *     return data["key"]
 * KeyError: 'key'
 */
export function parsePythonTraceback(text: string): ParsedError | null {
  const TB_START = /Traceback \(most recent call last\):/;
  const FRAME_RE = /^\s+File "(.+?)", line (\d+), in (.+)/;
  const CODE_RE = /^\s{4,}(.+)/;
  const ERROR_RE = /^(\w+(?:\.\w+)*):\s*(.*)$/;

  const match = text.match(TB_START);
  if (!match) return null;

  const lines = text.slice(match.index!).split('\n');
  const frames: StackFrame[] = [];
  let errorType = '';
  let errorMessage = '';
  let currentFrame: Partial<StackFrame> | null = null;

  for (let i = 1; i < lines.length; i++) {
    const line = lines[i];
    
    const frameMatch = line.match(FRAME_RE);
    if (frameMatch) {
      if (currentFrame?.file) frames.push(currentFrame as StackFrame);
      currentFrame = {
        file: frameMatch[1],
        line: parseInt(frameMatch[2], 10),
        function: frameMatch[3],
        code: '',
      };
      continue;
    }

    const codeMatch = line.match(CODE_RE);
    if (codeMatch && currentFrame) {
      currentFrame.code = codeMatch[1].trim();
      continue;
    }

    const errorMatch = line.match(ERROR_RE);
    if (errorMatch) {
      if (currentFrame?.file) frames.push(currentFrame as StackFrame);
      errorType = errorMatch[1];
      errorMessage = errorMatch[2];
      break;
    }
  }

  if (!errorType) return null;

  return {
    type: errorType,
    message: errorMessage,
    frames,
    raw: text,
    timestamp: Date.now(),
  };
}

/**
 * 也解析 Node.js 的 Error stack（bonus）
 * 
 * Error: Cannot find module 'foo'
 *     at Function.Module._resolveFilename (node:internal/modules/cjs/loader:1075:15)
 *     at /app/server.js:42:10
 */
export function parseNodeError(text: string): ParsedError | null {
  const ERROR_START = /^(\w+Error):\s*(.*)/m;
  const FRAME_RE = /^\s+at\s+(?:(.+?)\s+\()?(.+?):(\d+)(?::(\d+))?\)?$/;

  const errorMatch = text.match(ERROR_START);
  if (!errorMatch) return null;

  const lines = text.split('\n');
  const frames: StackFrame[] = [];

  for (const line of lines) {
    const frameMatch = line.match(FRAME_RE);
    if (frameMatch) {
      frames.push({
        file: frameMatch[2],
        line: parseInt(frameMatch[3], 10),
        function: frameMatch[1] || '<anonymous>',
        code: '',
      });
    }
  }

  return {
    type: errorMatch[1],
    message: errorMatch[2],
    frames,
    raw: text,
    timestamp: Date.now(),
  };
}
```

#### §1 cli/ 新增 detect-env.ts

```ts
import { execSync } from 'child_process';

export interface RuntimeEnv {
  language: string;       // 'python' | 'node' | 'unknown'
  version: string;        // '3.11.5'
  os: string;             // 'Windows 11' | 'macOS 14.5' | 'Linux 6.1'
  packages: string[];     // ['django==4.2', 'requests==2.31', ...]
}

export function detectRuntime(command: string): RuntimeEnv {
  const env: RuntimeEnv = {
    language: 'unknown',
    version: '',
    os: `${process.platform} ${process.arch}`,
    packages: [],
  };

  // 偵測語言
  if (/python|django|flask|gunicorn|uvicorn|celery/i.test(command)) {
    env.language = 'python';
    try { env.version = execSync('python --version 2>&1').toString().trim(); } catch {}
    try {
      const pipList = execSync('pip list --format=freeze 2>/dev/null || pip3 list --format=freeze 2>/dev/null')
        .toString().trim().split('\n').slice(0, 50); // 最多 50 個
      env.packages = pipList;
    } catch {}
  } else if (/node|npm|npx|ts-node|next|nest|express/i.test(command)) {
    env.language = 'node';
    try { env.version = execSync('node --version 2>&1').toString().trim(); } catch {}
    try {
      const pkg = execSync('npm list --depth=0 --json 2>/dev/null').toString();
      const deps = JSON.parse(pkg).dependencies || {};
      env.packages = Object.entries(deps).slice(0, 50).map(([k, v]: any) => `${k}@${v.version || '?'}`);
    } catch {}
  } else if (/go\s+run|go\s+build/i.test(command)) {
    env.language = 'go';
    try { env.version = execSync('go version 2>&1').toString().trim(); } catch {}
  }

  return env;
}
```

#### §1 Server：新增 GET /reports 頁面

需要登入才能看（從 session token 取 user_id）。用 `GET /api/reports` 拿資料（PM-132 已有認證 + user 過濾），server 端渲染 HTML。

```ts
async function reportsPage(request: Request, env: Env): Promise<Response> {
  const lang = getLang(request);
  const t = (zh: string, en: string) => lang === 'zh' ? zh : en;
  
  // 從 cookie 或 query 取 session token（報告頁是網頁，不是 extension）
  const url = new URL(request.url);
  const token = url.searchParams.get('token') || '';
  
  if (!token) {
    // 沒有 token → 顯示提示頁「請從 BugEzy 擴充開啟」
    return htmlWithCSP(`
      <h1>${t('📋 我的報告', '📋 My Reports')}</h1>
      <p>${t('請從 BugEzy Chrome 擴充的「我的報告」按鈕開啟此頁面。', 'Please open this page from the "My Reports" button in the BugEzy Chrome extension.')}</p>
    `, request);
  }

  // 驗證 token
  const userId = await verifySessionToken(token, env);
  if (!userId) {
    return htmlWithCSP(`<h1>${t('登入已過期', 'Session expired')}</h1><p>${t('請重新從擴充開啟。', 'Please reopen from the extension.')}</p>`, request);
  }

  // 查詢報告列表
  const { data: reports } = await supa(env)
    .from('reports')
    .select('report_id, url, title, description, created_at, console_count, network_count, voice_count, screenshot_count, rrweb_count')
    .eq('user_id', userId)
    .order('created_at', { ascending: false })
    .limit(50);

  // 渲染列表
  let rows = '';
  if (!reports || reports.length === 0) {
    rows = `<tr><td colspan="7" style="text-align:center;color:#888;padding:20px;">${t('還沒有報告，去錄製你的第一個 Bug 吧！', 'No reports yet. Record your first bug!')}</td></tr>`;
  } else {
    reports.forEach((r: any) => {
      const date = new Date(r.created_at).toLocaleString(lang === 'zh' ? 'zh-TW' : 'en-US');
      const title = r.title || r.url || t('未命名', 'Untitled');
      const desc = r.description ? r.description.slice(0, 50) : '';
      const badges = [
        r.console_count > 0 ? `❌${r.console_count}` : '',
        r.network_count > 0 ? `🌐${r.network_count}` : '',
        r.voice_count > 0 ? '🎙️' : '',
        r.screenshot_count > 0 ? '📸' : '',
        r.rrweb_count > 0 ? '🎬' : '',
      ].filter(Boolean).join(' ');
      
      rows += `<tr>
        <td>${date}</td>
        <td><a href="/report/${r.report_id}" target="_blank">${esc(title)}</a></td>
        <td>${esc(desc)}</td>
        <td>${badges}</td>
        <td><a href="/report/${r.report_id}" target="_blank">${t('查看', 'View')}</a></td>
      </tr>`;
    });
  }

  const html = `
    <h1>${t('📋 我的報告', '📋 My Reports')}</h1>
    <p>${t('共 ' + (reports?.length || 0) + ' 份報告', (reports?.length || 0) + ' reports')}</p>
    <table class="reports-table">
      <thead>
        <tr>
          <th>${t('時間', 'Time')}</th>
          <th>${t('標題 / 頁面', 'Title / Page')}</th>
          <th>${t('描述', 'Description')}</th>
          <th>${t('內容', 'Content')}</th>
          <th>${t('操作', 'Action')}</th>
        </tr>
      </thead>
      <tbody>${rows}</tbody>
    </table>
  `;

  return htmlWithCSP(html, request);
}
```
