#### §3 background.ts 改動

在 `RECORDING_DONE` case 裡，收到 summary 後追加上傳邏輯：

```ts
case 'RECORDING_DONE': {
  const summary = (msg as any).summary as RecordingSummary;
  // 先標記上傳中
  summary.uploadStatus = 'uploading';
  summary.shareUrl = null;
  summary.uploadError = null;
  const s = await setState({
    recording: false,
    startedAt: null,
    tabId: null,

  });
```

新增 `uploadPayload()` 函式：
```ts
async function uploadPayload(): Promise<void> {
  try {
    const r = await chrome.storage.local.get(STORAGE_KEY);
    const payload = r[STORAGE_KEY];
    if (!payload) {
      blog('uploadPayload: 沒有 payload，跳過');
      return;
    }

    blog('uploadPayload: 開始上傳...');
    const res = await fetch(`${API_BASE}/api/reports`, {
      method: 'POST',
      headers: { 'Content-Type': 'application/json' },
      body: JSON.stringify(payload),
    });

    if (!res.ok) {
      const errText = await res.text();
      throw new Error(`HTTP ${res.status}: ${errText}`);
    }

    const data = await res.json() as { report_id: string; share_url: string };
    blog('uploadPayload: 上傳成功', data.share_url);

    // 更新 state 裡的 summary
    const stateResult = await chrome.storage.local.get(STATE_KEY);
    const state = stateResult[STATE_KEY];
    if (state?.summary) {
      state.summary.uploadStatus = 'success';
      state.summary.shareUrl = data.share_url;
      state.summary.uploadError = null;
      await chrome.storage.local.set({ [STATE_KEY]: state });
    }
  } catch (err) {
    blog('uploadPayload: 上傳失敗', err);
    const stateResult = await chrome.storage.local.get(STATE_KEY);
    const state = stateResult[STATE_KEY];
    if (state?.summary) {
      state.summary.uploadStatus = 'error';
      state.summary.uploadError = String(err);
      await chrome.storage.local.set({ [STATE_KEY]: state });
    }
  }
}
```

**import 加上**：`API_BASE` from types。

#### §6 popup.ts 改動

`render()` 裡，根據 `state.summary.uploadStatus` 顯示不同狀態：

```ts
// 上傳狀態
const uploadArea = $('uploadArea');
const uploadStatusEl = $('uploadStatus');
const shareUrlRow = $('shareUrlRow');
const shareLink = $<HTMLAnchorElement>('shareLink');
const copyLinkBtn = $('copyLinkBtn');

function renderUpload(summary: RecordingSummary) {
  switch (summary.uploadStatus) {
    case 'uploading':
      uploadStatusEl.textContent = '⏳ 正在上傳到雲端...';
      shareUrlRow.style.display = 'none';
      break;
    case 'success':
      uploadStatusEl.textContent = '✅ 已上傳';
      if (summary.shareUrl) {
        shareUrlRow.style.display = 'flex';
        shareLink.href = summary.shareUrl;
        shareLink.textContent = summary.shareUrl;
      }
      break;
    case 'error':
      uploadStatusEl.textContent = '❌ 上傳失敗（可手動匯出 JSON）';
      shareUrlRow.style.display = 'none';
      break;
    default:
      uploadStatusEl.textContent = '';
      shareUrlRow.style.display = 'none';
  }
}

// 複製連結按鈕
copyLinkBtn.addEventListener('click', () => {
  const url = shareLink.href;
  if (url && url !== '#') {
    navigator.clipboard.writeText(url);
    copyLinkBtn.textContent = '✅ 已複製';
    setTimeout(() => { copyLinkBtn.textContent = '📋 複製連結'; }, 1500);
  }
});
```

在 `render()` 函式裡呼叫 `renderUpload(state.summary)` 顯示狀態。

**上傳中 → 輪詢更新**：因為上傳是 background 非同步做的，popup 打開時可能還在上傳中。加一個 polling：
```ts
// 如果正在上傳，每秒重新讀取狀態
if (state.summary?.uploadStatus === 'uploading') {
  const poll = setInterval(async () => {
    const fresh = await send<StateResponse>('GET_STATE');
    renderUpload(fresh.summary);
    if (fresh.summary?.uploadStatus !== 'uploading') {
      clearInterval(poll);
    }
  }, 1000);
}
```

#### §5 RrwebPlayer 元件（`web/src/components/RrwebPlayer.tsx`）

用 `rrweb-player` 套件。它不是 React 元件，是 vanilla JS class，需要用 `useRef` + `useEffect` 掛載：

```tsx
import { useRef, useEffect } from 'react';
import rrwebPlayer from 'rrweb-player';
import 'rrweb-player/dist/style.css';

interface Props {
  events: unknown[];
}

export default function RrwebReplay({ events }: Props) {
  const containerRef = useRef<HTMLDivElement>(null);

  useEffect(() => {
    if (!containerRef.current || events.length === 0) return;

    // 清空（防止 re-render 重複掛載）
    containerRef.current.innerHTML = '';

    new rrwebPlayer({
      target: containerRef.current,
      props: {
        events,
        width: 800,
        height: 450,
        autoPlay: false,
        showController: true,
        speedOption: [1, 2, 4],
      },
    });
  }, [events]);

  return <div ref={containerRef} />;
}
```

**易踩雷**：
- `rrweb-player` 的 CSS 必須 import，否則控制列不顯示
- `events` 必須是 rrweb 格式的陣列（我們的 payload 就是，直接傳）
- player 的 `target` 是 DOM 元素，不是 React ref value，要用 `.current`
- 寬高先固定 800×450，日後可做響應式

#### 三處小強化（行為等價/更穩，已說明）

#### §2 重寫 `web/src/components/RrwebPlayer.tsx`

整個檔案替換：

```tsx
import { useRef, useEffect, useState } from 'react';
import { Replayer } from '@rrweb/replay';
import '@rrweb/replay/dist/style.css';

interface Props {
  events: unknown[];
}

export default function RrwebReplay({ events }: Props) {
  const containerRef = useRef<HTMLDivElement>(null);
  const replayerRef = useRef<Replayer | null>(null);
  const [playing, setPlaying] = useState(false);
  const [progress, setProgress] = useState(0);
  const [duration, setDuration] = useState(0);
  const rafRef = useRef<number>(0);

  useEffect(() => {
    const target = containerRef.current;
    if (!target || events.length < 2) return;

    // 清空舊的
    target.innerHTML = '';

    try {
      const replayer = new Replayer(events as any, {
        root: target,
        skipInactive: true,
        showWarning: false,
        liveMode: false,
      });

      replayerRef.current = replayer;

      // 取得總時長
      const meta = replayer.getMetaData();
      const dur = meta.totalTime || 0;
      setDuration(dur);

      // 進度更新 loop
      const updateProgress = () => {
        if (replayerRef.current) {
          const current = replayerRef.current.getCurrentTime();
          setProgress(current);
          if (current < dur) {
            rafRef.current = requestAnimationFrame(updateProgress);
          } else {
            setPlaying(false);
          }
        }
      };

      // Replayer 內部 iframe 設定固定尺寸
      const iframe = target.querySelector('iframe');
      if (iframe) {
        iframe.style.width = '800px';
        iframe.style.height = '450px';
        iframe.style.border = '1px solid #333';
        iframe.style.borderRadius = '4px';
        iframe.style.transform = 'scale(1)';
        iframe.style.transformOrigin = 'top left';
      }

      return () => {
        cancelAnimationFrame(rafRef.current);
        replayer.pause();
        replayerRef.current = null;
        target.innerHTML = '';
      };
    } catch (err) {
      console.error('[BugEzy] Replayer 建立失敗:', err);
      target.innerHTML = '<div class="empty">DOM 回放載入失敗</div>';
    }
  }, [events]);

  if (events.length < 2) {
    return <div className="empty">DOM 軌跡不足，無法回放（{events.length} 筆）</div>;
  }

  const handlePlay = () => {
    const r = replayerRef.current;
    if (!r) return;
    if (playing) {
      r.pause();
      cancelAnimationFrame(rafRef.current);
      setPlaying(false);
    } else {
      // 如果已播完，從頭開始
      if (progress >= duration - 100) {
        r.play(0);
      } else {
        r.resume(progress);
      }
      setPlaying(true);
      // 開始追蹤進度
      const update = () => {
        if (replayerRef.current) {
          const current = replayerRef.current.getCurrentTime();
          setProgress(current);
          if (current < duration) {
            rafRef.current = requestAnimationFrame(update);
          } else {
            setPlaying(false);
          }
        }
      };
      rafRef.current = requestAnimationFrame(update);
    }
  };

  const handleSeek = (e: React.ChangeEvent<HTMLInputElement>) => {
    const time = Number(e.target.value);
    const r = replayerRef.current;
    if (!r) return;
    r.play(time);
    if (!playing) {
      // 跳到位置後暫停
      setTimeout(() => r.pause(time), 50);
    }
    setProgress(time);
  };

  const fmtTime = (ms: number) => {
    const s = Math.floor(ms / 1000);
    const m = Math.floor(s / 60);
    const sec = s % 60;
    return `${m}:${String(sec).padStart(2, '0')}`;
  };

  return (
    <div>
      <div ref={containerRef} style={{ overflow: 'hidden', maxWidth: '800px' }} />
      <div className="player-controls">
        <button onClick={handlePlay} className="play-btn">
          {playing ? '⏸ 暫停' : '▶ 播放'}
        </button>
        <input
          type="range"
          min={0}
          max={duration}
          value={progress}
          onChange={handleSeek}
          style={{ flex: 1, margin: '0 12px' }}
        />
        <span className="time-display">
          {fmtTime(progress)} / {fmtTime(duration)}
        </span>
      </div>
    </div>
  );
}
```

```ts
function createMcpServer(env: Env): McpServer {
  const server = new McpServer({
    name: 'BugEzy',
    version: '0.1.0',
  });

  const supabase = () => createClient(env.SUPABASE_URL, env.SUPABASE_ANON_KEY);

  // Tool 1: list_reports
  server.tool('list_reports', '列出最近的 Bug 報告', {
    limit: { type: 'number', description: '筆數（1-50，預設 10）' },
    url: { type: 'string', description: 'URL 模糊搜尋（選填）' },
  }, async (args) => {
    const limit = Math.min(Math.max(Number(args.limit) || 10, 1), 50);
    let query = supabase()
      .from('reports')
      .select('report_id, url, title, browser, screen_size, console_count, network_count, voice_count, rrweb_count, created_at')
      .order('created_at', { ascending: false })
      .limit(limit);
    if (args.url) query = query.ilike('url', `%${args.url}%`);
    const { data, error } = await query;
    if (error) return { content: [{ type: 'text', text: `查詢失敗: ${error.message}` }] };
    return { content: [{ type: 'text', text: JSON.stringify(data, null, 2) }] };
  });

  // Tool 2: get_report_overview
  server.tool('get_report_overview', '取得報告概覽（metadata，不含原始資料）', {
    report_id: { type: 'string', description: '報告 ID' },
  }, async (args) => {
    const { data, error } = await supabase()
      .from('reports')
      .select('report_id, url, title, browser, screen_size, console_count, network_count, voice_count, rrweb_count, created_at')
      .eq('report_id', args.report_id)
      .single();
    if (error || !data) return { content: [{ type: 'text', text: '找不到報告' }] };
    return { content: [{ type: 'text', text: JSON.stringify(data, null, 2) }] };
  });

  // Tool 3: get_console_logs
  server.tool('get_console_logs', '取得 Console 記錄（warn/error）', {
    report_id: { type: 'string', description: '報告 ID' },
  }, async (args) => {
    const { data, error } = await supabase()
      .from('reports')
      .select('console_logs')
      .eq('report_id', args.report_id)
      .single();
    if (error || !data) return { content: [{ type: 'text', text: '找不到報告' }] };
    return { content: [{ type: 'text', text: JSON.stringify(data.console_logs, null, 2) }] };
  });

  // Tool 4: get_network_errors
  server.tool('get_network_errors', '取得 Network 錯誤（4xx/5xx）', {
    report_id: { type: 'string', description: '報告 ID' },
  }, async (args) => {
    const { data, error } = await supabase()
      .from('reports')
      .select('network_errors')
      .eq('report_id', args.report_id)
      .single();
    if (error || !data) return { content: [{ type: 'text', text: '找不到報告' }] };
    return { content: [{ type: 'text', text: JSON.stringify(data.network_errors, null, 2) }] };
  });

  // Tool 5: get_voice_transcript — 最有價值的除錯線索
  server.tool('get_voice_transcript', '取得開發者語音描述（中文轉錄）', {
    report_id: { type: 'string', description: '報告 ID' },
  }, async (args) => {
    const { data, error } = await supabase()
      .from('reports')
      .select('voice_transcript')
      .eq('report_id', args.report_id)
      .single();
    if (error || !data) return { content: [{ type: 'text', text: '找不到報告' }] };
    return { content: [{ type: 'text', text: JSON.stringify(data.voice_transcript, null, 2) }] };
  });

  // Tool 6: get_page_info
  server.tool('get_page_info', '取得頁面資訊（URL/標題/瀏覽器）', {
    report_id: { type: 'string', description: '報告 ID' },
  }, async (args) => {
    const { data, error } = await supabase()
      .from('reports')
      .select('url, title, browser, screen_size, created_at')
      .eq('report_id', args.report_id)
      .single();
    if (error || !data) return { content: [{ type: 'text', text: '找不到報告' }] };
    return { content: [{ type: 'text', text: JSON.stringify(data, null, 2) }] };
  });

  // Tool 7: get_rrweb_summary（從 R2 讀，只回摘要）
  server.tool('get_rrweb_summary', 'DOM 軌跡摘要（事件數/時長/類型分布，不回完整資料）', {
    report_id: { type: 'string', description: '報告 ID' },
  }, async (args) => {
    const { data: meta } = await supabase()
      .from('reports')
      .select('rrweb_r2_key, rrweb_count')
      .eq('report_id', args.report_id)
      .single();
    if (!meta?.rrweb_r2_key) return { content: [{ type: 'text', text: '無 DOM 軌跡' }] };
    const obj = await env.R2.get(meta.rrweb_r2_key);
    if (!obj) return { content: [{ type: 'text', text: 'R2 檔案不存在' }] };
    const events: any[] = JSON.parse(await obj.text());
    const types: Record<string, number> = {};
    events.forEach(e => { types[`type_${e.type}`] = (types[`type_${e.type}`] || 0) + 1; });
    const timestamps = events.map(e => e.timestamp).filter(Boolean);
    const duration = timestamps.length >= 2 ? Math.max(...timestamps) - Math.min(...timestamps) : 0;
    return { content: [{ type: 'text', text: JSON.stringify({ event_count: events.length, duration_ms: duration, event_types: types }, null, 2) }] };
  });

  // Tool 8: get_rrweb_events（完整資料，⚠ 大）
  server.tool('get_rrweb_events', '取得完整 DOM 事件（⚠ 資料量大）', {
    report_id: { type: 'string', description: '報告 ID' },
  }, async (args) => {
    const { data: meta } = await supabase()
      .from('reports')
      .select('rrweb_r2_key')
      .eq('report_id', args.report_id)
      .single();
    if (!meta?.rrweb_r2_key) return { content: [{ type: 'text', text: '無 DOM 軌跡' }] };
    const obj = await env.R2.get(meta.rrweb_r2_key);
    if (!obj) return { content: [{ type: 'text', text: 'R2 檔案不存在' }] };
    return { content: [{ type: 'text', text: await obj.text() }] };
  });

  return server;
}
```

**annotate.ts**：標注畫布邏輯

核心功能：
1. 從 URL query param `screenshotKey` 讀取 chrome.storage.local 裡的暫存截圖
2. 把截圖畫到 canvas 上作為背景
3. 支援四種工具：畫筆（freehand）、箭頭、框框、文字
4. 復原（undo）：用 history stack 存每一步
5. 完成：把 canvas 轉 dataUrl → 送回 background → 關閉此分頁

```ts
// ── 初始化 ──
const params = new URLSearchParams(location.search);
const key = params.get('key'); // 暫存 key
// 從 storage 讀截圖 base64
const data = await chrome.storage.local.get(key);
const img = new Image();
img.src = data[key]; // base64 dataUrl
await img.decode();
canvas.width = img.width;
canvas.height = img.height;
ctx.drawImage(img, 0, 0); // 畫底圖

// ── 工具邏輯 ──
let currentTool: 'pen' | 'arrow' | 'rect' | 'text' = 'pen';
let drawing = false;
let history: ImageData[] = []; // undo stack
let startX = 0, startY = 0;

// 每次操作前 push 當前 canvas 到 history
function saveState() {
  history.push(ctx.getImageData(0, 0, canvas.width, canvas.height));
  if (history.length > 50) history.shift(); // 限制 50 步
}

// mousedown/mousemove/mouseup 處理各工具
// pen: moveTo → lineTo (freehand)
// arrow: 畫線 + 三角箭頭
// rect: strokeRect
// text: prompt 輸入文字 → fillText

// ── 儲存 ──
saveBtn.onclick = async () => {
  const annotatedDataUrl = canvas.toDataURL('image/png');
  // 送回 background 存入截圖陣列
  await chrome.runtime.sendMessage({
    type: 'SCREENSHOT_ANNOTATED',
    dataUrl: annotatedDataUrl,
    timestamp: Number(params.get('timestamp')),
  });
  window.close();
};

cancelBtn.onclick = () => {
  // 取消：不存，刪除暫存
  chrome.storage.local.remove(key!);
  window.close();
};
```

#### annotate.ts 畫布核心

- **載入**：URL `key` → `chrome.storage.local` 讀暫存 base64 → `img.decode()` → `drawImage` 當底圖，存 `baseImage`（清除全部用）。
- **4 工具**：

  - ➡️ 箭頭：拖曳預覽（mousedown 存 `snapshot` → mousemove `putImageData(snapshot)` 重畫）；`drawArrow()` = 線 + `atan2` 算角度的三角箭頭（fill）。
  - ⬜ 框框：同預覽機制，`strokeRect(startX,startY,dx,dy)`。
  - 📝 文字：`prompt()` 取字 → `fillText`，字級隨粗細（12+width*4）。
- **undo**：每次操作前 `saveState()` push `ImageData`（上限 50 步），復原 `pop()` + `putImageData`。

- **儲存**：`canvas.toDataURL('image/png')` → `sendMessage(SCREENSHOT_ANNOTATED)` → `window.close()`；**取消**：刪暫存 key → close。
- 座標用 `getBoundingClientRect` + `canvas.width/rect.width` 比例換算，避免 CSS 縮放偏移。

#### §4 區域截圖的捲動拼接邏輯

這是核心難點。使用者的起點和終點可能跨越多個 viewport 高度。

**步驟**：
```
1. 拿到起點 (x1, y1) 和終點 (x2, y2)（document 絕對座標）
2. 計算選取矩形：left = min(x1,x2), top = min(y1,y2), right = max(x1,x2), bottom = max(y1,y2)
3. 計算需要幾段 viewport 擷取：
   - viewportHeight = window.innerHeight
   - segments = ceil((bottom - top) / viewportHeight) + 1（多取一段避免邊界）
4. 建一個大 canvas：width = viewport.width, height = (bottom - top)
5. 移除遮罩（擷取前要移掉，否則截圖會包含遮罩）
6. for each segment:
   a. window.scrollTo(0, top + segment * viewportHeight)
   b. 等一幀（requestAnimationFrame）
   c. 送 CAPTURE_SEGMENT 給 background → background 呼叫 captureVisibleTab() → 回傳 dataUrl
   d. 把 dataUrl 畫到大 canvas 的對應位置
7. 恢復原始 scroll 位置
8. 從大 canvas 裁切出選取矩形
9. 把裁切後的 dataUrl 送給 background → 開標注頁
```

**CAPTURE_SEGMENT 訊息**（新增，content ↔ background 通訊）：
```ts
// content → background
{ type: 'CAPTURE_SEGMENT' }
// background 回應
{ dataUrl: string }
```

background handler：
```ts
case 'CAPTURE_SEGMENT': {
  const dataUrl = await chrome.tabs.captureVisibleTab(undefined, { format: 'png' });
  sendResponse({ dataUrl });
  break;
}
```

**⚠ 易踩雷**：
- `captureVisibleTab` 需要 `return true` 保持 sendResponse 活著（非同步）
- 擷取前要**移除 overlay DOM**，擷取後**恢復**
- 多段擷取之間要 `await new Promise(r => requestAnimationFrame(r))` 等頁面渲染完
- 如果選取區域在一個 viewport 內（不需捲動），直接單次擷取 + 裁切即可

#### 難點處理（區域跨 viewport 拼接）

- 起點/終點記 **document 絕對座標**（`clientX + scrollX`）；預覽矩形即時換算回 viewport 座標，捲動時自動跟隨。

- 大 canvas 以 **device pixel** 為單位（`vw*dpr` × `rectH*dpr`）；每段以**實際 `window.scrollY`**（非目標值）算貼上位置，自動處理底部 clamp/重疊；`guard<40` 防無限迴圈；結束 `scrollTo` 還原原始捲動位置；最後依 `left*dpr` 水平裁切出選取區。

#### §2 annotate.ts 語音輸入邏輯

```ts
const descInput = document.getElementById('descInput') as HTMLTextAreaElement;
const voiceInputBtn = document.getElementById('voiceInputBtn') as HTMLButtonElement;
const voiceStatus = document.getElementById('voiceStatus') as HTMLDivElement;

let recognition: any = null;
let listening = false;

const SR = (window as any).SpeechRecognition || (window as any).webkitSpeechRecognition;

voiceInputBtn.addEventListener('click', () => {
  if (!SR) {
    voiceStatus.textContent = '此瀏覽器不支援語音辨識';
    return;
  }

  if (listening) {
    // 停止
    recognition?.stop();
    listening = false;
    voiceInputBtn.classList.remove('listening');
    voiceInputBtn.textContent = '🎤';
    voiceStatus.textContent = '';
    return;
  }

  // 開始聽
  recognition = new SR();
  recognition.lang = 'zh-TW';
  recognition.continuous = true;
  recognition.interimResults = true;

  let finalTranscript = '';

  recognition.onresult = (e: any) => {
    let interim = '';
    for (let i = e.resultIndex; i < e.results.length; i++) {
      const text = e.results[i][0].transcript;
      if (e.results[i].isFinal) {
        finalTranscript += text;
      } else {
        interim = text;
      }
    }
    // 即時更新：已確認的文字寫入 textarea，interim 顯示在狀態列
    if (finalTranscript) {
      descInput.value = descInput.value + finalTranscript;
      finalTranscript = '';
    }
    voiceStatus.textContent = interim ? `🔴 辨識中：${interim}` : '🔴 聆聽中...';
  };

  recognition.onend = () => {
    // Web Speech API 會自動停，如果還在聽就重啟
    if (listening) {
      try { recognition.start(); } catch (_) {}
    }
  };

  recognition.onerror = (e: any) => {
    voiceStatus.textContent = `語音錯誤：${e.error}`;
    if (e.error === 'not-allowed') {
      listening = false;
      voiceInputBtn.classList.remove('listening');
      voiceInputBtn.textContent = '🎤';
    }
  };

  recognition.start();
  listening = true;
  voiceInputBtn.classList.add('listening');
  voiceInputBtn.textContent = '⏹';
  voiceStatus.textContent = '🔴 聆聽中...';
});
```

**重點**：
- `interimResults = true`（即時預覽，讓使用者看到辨識中的文字）
- 確認的文字直接 append 到 textarea
- 按 🎤 開始，再按 ⏹ 停止（toggle）
- annotate.html 是擴充頁面（chrome-extension:// origin），麥克風權限歸擴充。第一次會彈授權，授權後記住。
- `continuous = true`，自動重啟 onend

#### §2 annotate.ts 改動

**A. 抽出 startListening() / stopListening()：**

```ts
const captionText = document.getElementById('captionText')!;
const liveCaptions = document.getElementById('liveCaptions')!;

function startListening() {
  if (!SR || listening) return;

  recognition = new SR();
  recognition.lang = 'zh-TW';
  recognition.continuous = true;
  recognition.interimResults = true;

  recognition.onresult = (e: any) => {
    let interim = '';
    for (let i = e.resultIndex; i < e.results.length; i++) {
      const text = e.results[i][0].transcript;
      if (e.results[i].isFinal) {
        // 確定的文字 → 寫入文字框
        descInput.value += text;
        // 字幕顯示確認
        captionText.textContent = '✅ ' + text;
        setTimeout(() => {
          if (listening) captionText.textContent = '🔴 聆聽中...';
        }, 1500);
      } else {
        interim = text;
      }
    }
    // interim → 即時字幕（使用者正在講的，還沒確定）
    if (interim) {
      captionText.textContent = '🔴 ' + interim;
      liveCaptions.classList.remove('hidden');
    }
  };

  recognition.onend = () => {
    if (listening) {
      try { recognition.start(); } catch (_) {}
    }
  };

  recognition.onerror = (e: any) => {
    captionText.textContent = '語音錯誤：' + e.error;
    if (e.error === 'not-allowed') {
      listening = false;
      voiceInputBtn.classList.remove('listening');
      voiceInputBtn.textContent = '🎤';
      liveCaptions.classList.add('hidden');
    }
  };

  recognition.start();
  listening = true;
  voiceInputBtn.classList.add('listening');
  voiceInputBtn.textContent = '⏹';
  voiceStatus.textContent = '';
  captionText.textContent = '🔴 聆聽中，邊畫邊說描述問題...';
  liveCaptions.classList.remove('hidden');
}

function stopListening() {
  if (!listening) return;
  recognition?.stop();
  listening = false;
  voiceInputBtn.classList.remove('listening');
  voiceInputBtn.textContent = '🎤';
  voiceStatus.textContent = '語音已停止';
  liveCaptions.classList.add('hidden');
}
```

### 🟡 PM-23：修標注頁語音中斷（工具按鈕不搶焦點）

**問題**：在標注頁畫箭頭或切換工具時，語音辨識會中斷。原因是點擊按鈕觸發焦點切換，瀏覽器的 SpeechRecognition 被打斷。

**修法**：所有工具列按鈕加 `mousedown` + `e.preventDefault()`，阻止按鈕搶走焦點。

---

#### 改動範圍：只動 `extension/src/annotate.ts`（和 annotate.html 如果按鈕是在 html 裡定義的話）

---

#### 改動內容

**對所有工具列按鈕**（畫筆、箭頭、框框、文字、顏色、粗細、復原、清除），加上：

```ts
button.addEventListener('mousedown', (e) => {
  e.preventDefault(); // 阻止按鈕搶焦點，避免打斷 SpeechRecognition
});
```

**具體做法**：在 annotate.ts 裡找到所有工具列按鈕的初始化區域，對每個按鈕都加上 `mousedown preventDefault`。

最簡潔的寫法——在工具列容器層級攔截：

```ts
// 在 toolbar DOM 建好之後加這一行
const toolbar = document.querySelector('.toolbar');
if (toolbar) {
  toolbar.addEventListener('mousedown', (e) => {
    // 阻止工具列裡所有按鈕、select、label 搶焦點
    // 但不擋 color input 和 select（它們需要焦點來展開選單）
    const target = e.target as HTMLElement;
    const tag = target.tagName.toLowerCase();
    if (tag !== 'input' && tag !== 'select') {
      e.preventDefault();
    }
  });
}
```

**為什麼這樣做**：
- `preventDefault()` 在 `mousedown` 上會阻止瀏覽器的預設行為（搶焦點）
- `click` 事件仍然會觸發（`preventDefault` 不阻止 click）
- 顏色選擇器（`<input type="color">`）和粗細下拉（`<select>`）排除在外，它們需要焦點才能正常運作
- 這樣語音辨識就不會被工具切換打斷

---

#### 額外改進：縮短 onend 重啟延遲

在 `recognition.onend` 裡，確認重啟時用最短延遲：

```ts
recognition.onend = () => {
  if (listening) {
    // 立即重啟，不等
    try { recognition.start(); } catch (_) {}
  }
};
```

如果 `recognition.start()` 在 `onend` 裡直接呼叫報錯（有些 Chrome 版本會），加 `setTimeout(() => recognition.start(), 50)` 最小延遲。
